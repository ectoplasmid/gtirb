167  bool removeModule(Module* M) {
168  auto& Index = Modules.get<by_pointer>();
169  if (auto Iter = Index.find(M); Iter != Index.end()) {
170  MO->removeProxyBlocks(M, M->proxy_blocks());
171  MO->removeCodeBlocks(M, M->code_blocks());
172  Index.erase(Iter);
173  M->setParent(nullptr, nullptr);
174  return true;
175  }
176  return false;
177  }
178 
182  Module* addModule(Module* M) {
183  if (M->getIR()) {
184  M->getIR()->removeModule(M);
185  }
186 
187  MO->addProxyBlocks(M, M->proxy_blocks());
188  MO->addCodeBlocks(M, M->code_blocks());
189  Modules.emplace(M);
190  M->setParent(this, MO.get());
191  return M;
192  }
