51 template <typename ForwardIterator,
52  typename Compare = std::less<
53  typename std::iterator_traits<ForwardIterator>::value_type>>
54 class MergeSortedIterator
55  : public boost::iterator_facade<
56  MergeSortedIterator<ForwardIterator, Compare>,
57  typename std::iterator_traits<ForwardIterator>::value_type,
58  boost::forward_traversal_tag,
59  typename std::iterator_traits<ForwardIterator>::reference,
60  typename std::iterator_traits<ForwardIterator>::difference_type> {
61 public:
65  MergeSortedIterator() = default;
66 
73  template <typename RangeIteratorRange>
74  explicit MergeSortedIterator(RangeIteratorRange RangeRange) {
75  for (const auto& Range : RangeRange) {
76  if (auto RBegin = Range.begin(), REnd = Range.end(); RBegin != REnd) {
77  Ranges.emplace_back(RBegin, REnd);
78  }
79  }
80  // Establish the heap invariant for Ranges.
81  std::make_heap(Ranges.begin(), Ranges.end(), rangeGreaterThan);
82  }
83 
91  template <typename RangeIterator>
92  MergeSortedIterator(RangeIterator Begin, RangeIterator End)
93  : MergeSortedIterator(boost::make_iterator_range(Begin, End)) {}
94 
95  // Beginning of functions for iterator facade compatibility.
96  typename std::iterator_traits<ForwardIterator>::reference
97  dereference() const {
98  assert(!Ranges.empty() && "Attempt to dereference end of iterator!");
99  return *Ranges.front().first;
100  }
101 
102  bool equal(const MergeSortedIterator<ForwardIterator, Compare>& Other) const {
103  return Ranges == Other.Ranges;
104  }
105 
106  void increment() {
107  assert(!Ranges.empty() && "Attempt to increment end of iterator!");
108  // After incrementing the first range, it may no longer have the lowest
109  // first element. Removing the range, then re-inserting it ensures that the
110  // heap invariant is maintained.
111  std::pop_heap(Ranges.begin(), Ranges.end(), rangeGreaterThan);
112  ++Ranges.back().first;
113  if (Ranges.back().first == Ranges.back().second) {
114  Ranges.pop_back();
115  } else {
116  std::push_heap(Ranges.begin(), Ranges.end(), rangeGreaterThan);
117  }
118  }
119  // End of functions for iterator facade compatibility.
120 private:
121  using RangeType = std::pair<ForwardIterator, ForwardIterator>;
122 
123  // Compares two ranges according to the relationship of their first elements
124  // given by \c Compare. Empty ranges are treated as being greater than any
125  // non-empty range.
126  static bool rangeGreaterThan(const RangeType& R1, const RangeType& R2) {
127  if (R1.first == R1.second)
128  // An empty R1 is greater than every R2.
129  return true;
130  if (R2.first == R2.second)
131  // Any R1 is less than an empty R2.
132  return false;
133  // Flip the comparison to implement "greater-than".
134  return Compare()(*R2.first, *R1.first);
135  }
136 
137  // Ranges is a heap ordered by \c rangeGreaterThan. This ensures that the
138  // range with the lowest first element (according to \c Compare) is always at
139  // the front.
140  std::vector<RangeType> Ranges;
141 };
