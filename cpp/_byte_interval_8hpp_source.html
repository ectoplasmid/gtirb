1549  template <typename ByteIntervalType, typename T> class BytesReference {
1550  public:
1551  BytesReference(ByteIntervalType* BI_, size_t I_,
1552  boost::endian::order InputOrder_,
1553  boost::endian::order OutputOrder_)
1554  : BI(BI_), I(I_), InputOrder(InputOrder_), OutputOrder(OutputOrder_) {}
1555 
1561  operator T() const {
1562  assert(I + sizeof(T) <= BI->Size &&
1563  "read into interval's bytes out of bounds!");
1564 
1565  auto S = BI->Bytes.size();
1566 
1567  if (I >= S) {
1568  // anything this far past the end of initialized bytes is composed of
1569  // all zero bytes, so we return what T would have been interpreted as
1570  // if all bytes are zero.
1571  //
1572  // (note that you may be tempted to replace this with "return T{};",
1573  // but beware: T might be a non-scalar type whose default constructor
1574  // differs from the value returned when all bytes are re-interpeted as
1575  // zeroes. A similar argument exists for "return T{0};".)
1576  const std::array<uint8_t, sizeof(T)> Array{};
1577  return *reinterpret_cast<const T*>(Array.data());
1578  }
1579 
1580  if (I + sizeof(T) > S) {
1581  // Here, I < S < I + sizeof(T), so we need to partially fill the
1582  // initialized bytes and combine it with zeroes for the uninitialized
1583  // bytes.
1584  std::array<uint8_t, sizeof(T)> Array{};
1585  // Thanks to math, 0 < S - I < sizeof(T).
1586  std::copy_n(BI->Bytes.begin() + I, S - I, Array.begin());
1587  return boost::endian::conditional_reverse(
1588  *reinterpret_cast<const T*>(Array.data()), InputOrder, OutputOrder);
1589  }
1590 
1591  return boost::endian::conditional_reverse(
1592  *reinterpret_cast<const T*>(BI->Bytes.data() + I), InputOrder,
1593  OutputOrder);
1594  }
1595 
1602  BytesReference<ByteIntervalType, T>& operator=(const T& rhs) {
1603  assert(I + sizeof(T) <= BI->Size &&
1604  "write into interval's bytes out of bounds!");
1605 
1606  if (I + sizeof(T) > BI->Bytes.size()) {
1607  BI->Bytes.resize(I + sizeof(T));
1608  }
1609 
1610  *reinterpret_cast<T*>(BI->Bytes.data() + I) =
1611  boost::endian::conditional_reverse(rhs, OutputOrder, InputOrder);
1612  return *this;
1613  }
